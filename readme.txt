// ----------------------------- //
// ------RAPPELS BRANCHING------ //
// ----------------------------- //

- Ajouter une nouvelle branche -
git branch 'nomDeBranche'

- Aller sur la branche
git checkout 'nomDeBranche'

- Voir les branches -
git branch

- Récupérer les données depuis le dépôt distant -
git pull origin 'nomDeBranche'

- Fusionner la branche SUR LAQUELLE ON SE TROUVE avec une autre branche -
git merge 'nomDeBranche'

- Supprimer la branche -
git branch -d 'nomDeBranche'
